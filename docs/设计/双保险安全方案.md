# 双保险安全方案

AI Gateway 守前门 + Engine 级写入拦截守后门。

## 核心理念

| 操作 | 策略 |
|------|------|
| **读（SELECT）** | 完全自由，Agent 可用任意 SQL 做复杂分析 |
| **写（DML）** | 绝对管制，必须有令牌（令牌仅由白名单 API 发放） |
| **结构（DDL）** | 绝对禁止，Agent 永远不能改表结构 |
| **启动维护** | 特权豁免，`_maintenance_mode` 绕过全部检查 |

## data layer（database.py）

```python
import contextvars
from sqlalchemy import create_engine, event, text
from sqlalchemy.orm import sessionmaker, Session, DeclarativeBase
from sqlalchemy.sql.dml import Update, Delete, Insert

from workspace import get_db_path as _get_db_path

# ── 全局状态 ──
_request_write_perm = contextvars.ContextVar('_request_write_perm', default=False)
_maintenance_mode = False

def _is_write_permitted() -> bool:
    return _maintenance_mode or _request_write_perm.get()

def set_maintenance_mode(val: bool):
    global _maintenance_mode
    _maintenance_mode = val

# ── Engine 私有化（防 from database import engine; engine.connect().execute("DELETE...")）──
DB_PATH = _get_db_path()
DATABASE_URL = f"sqlite:///{DB_PATH}"
_engine = create_engine(DATABASE_URL, connect_args={"check_same_thread": False})

def get_db_url() -> str:
    """只读返回数据库 URL（取代 get_engine，防止引擎被用于 execute）"""
    return DATABASE_URL

def configure_engine(db_url: str):
    """测试用：替换 engine + 重建 SessionLocal"""
    global _engine, SessionLocal
    _engine = create_engine(db_url, connect_args={"check_same_thread": False})
    SessionLocal = sessionmaker(class_=SecureSession, bind=_engine)

# ── Engine 级拦截钩子（single source of truth）──
@event.listens_for(_engine, 'before_execute')
def _guard_execute(conn, clauseelement, multiparams, params, execution_options):
    if _maintenance_mode:
        return

    stmt_str = str(clauseelement).lower()
    first_word = stmt_str.strip().split()[0] if stmt_str.strip() else ""

    # DDL 红线：首单词检测（避免表名含关键字导致的误杀）
    if first_word in {"drop", "alter", "create", "truncate"}:
        raise BusinessError(code=ErrorCode.SECURITY_VIOLATION,
                            message="禁止执行结构变更(DDL)")

    # ORM-level DML：SQLAlchemy 对象直接类型检查
    if isinstance(clauseelement, (Update, Delete, Insert)):
        if not _is_write_permitted():
            raise BusinessError(code=ErrorCode.SECURITY_VIOLATION,
                                message="禁止直接写数据库，请调用合规 API")
        return

    # Raw SQL DML
    if isinstance(clauseelement, text):
        _guard_text_dml(stmt_str, first_word)


def _guard_text_dml(stmt_str: str, first_word: str):
    """text('...') 写操作拦截（覆盖 CTE 绕过）"""
    dml = {"insert", "update", "delete", "replace"}
    # 首单词检测（普通 DML）
    if first_word in dml and not _is_write_permitted():
        raise BusinessError(code=ErrorCode.SECURITY_VIOLATION,
                            message="禁止直接写数据库，请调用合规 API")
    # 全语句扫描（CTE: WITH ... DELETE ...）
    if first_word not in dml:
        import re
        for kw in dml:
            if re.search(rf'\b{kw}\b', stmt_str) and not _is_write_permitted():
                raise BusinessError(code=ErrorCode.SECURITY_VIOLATION,
                                    message="禁止直接写数据库（CTE 内 DML 被拦截）")

# ── SecureSession（ORM 级逻辑防线，Engine 钩子是物理防线）──
class SecureSession(Session):
    """connection() 不拦截，交给 Engine 钩子兜底"""

SessionLocal = sessionmaker(class_=SecureSession, bind=_engine)

@event.listens_for(SecureSession, 'before_flush')
def _guard_flush(session, context, instances):
    if _is_write_permitted():
        return
    if session.new or session.dirty or session.deleted:
        raise BusinessError(code=ErrorCode.SECURITY_VIOLATION,
                            message="ORM 操作被拦截：请通过 API 操作")


class Base(DeclarativeBase):
    pass


def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


def init_db():
    import models
    import models_finance
    Base.metadata.create_all(bind=_engine)
```

## 权限中间件（middleware/write_permission.py）

```python
from starlette.types import ASGIApp, Scope, Receive, Send
from database import _request_write_perm
from ai_gateway import _gateway_authorized

WRITE_METHODS = {"POST", "PUT", "DELETE", "PATCH"}

class WritePermissionMiddleware:
    def __init__(self, app: ASGIApp):
        self.app = app

    async def __call__(self, scope: Scope, receive: Receive, send: Send):
        if scope["type"] != "http":
            await self.app(scope, receive, send)
            return
        if scope["method"] in WRITE_METHODS and _gateway_authorized.get():
            token = _request_write_perm.set(True)
            try:
                await self.app(scope, receive, send)
            finally:
                _request_write_perm.reset(token)
        else:
            await self.app(scope, receive, send)
```

## AI Gateway 联动（ai_gateway.py）

```python
import contextvars
_gateway_authorized = contextvars.ContextVar('_gateway_authorized', default=False)

_SKIP_PREFIXES = {"/api/auth", "/api/bootstrap"}

# 白名单
CAPABILITIES = [
    Capability("POST", "/api/products", "创建商品"),
    Capability("POST", "/api/bank-accounts", "创建银行账户"),
    # ...
]

def dispatch(request):
    # 跳过路径（初始化、认证）
    if any(request.url.path.startswith(p) for p in _SKIP_PREFIXES):
        _gateway_authorized.set(True)
        return next_handler(request)

    operator = request.headers.get("X-Operator")

    # User 请求（登录后信任）
    if operator == "user":
        _gateway_authorized.set(True)
        return next_handler(request)

    # AI 请求（必须命中白名单）
    if operator == "ai" and _match_whitelist(request, CAPABILITIES):
        _gateway_authorized.set(True)
        return next_handler(request)

    raise PermissionError("无权访问")
```

@router.post("/init")
def bootstrap_init(db=Depends(get_db)):
    if db.query(Account).count() > 0:
        return {"status": "already", "account_id": db.query(Account).first().id}
    account = Account(name="默认企业账本", type="company", code="default")
    db.add(account); db.flush()
    ledger = Ledger(name="默认账本", code="default"); db.add(ledger); db.flush()
    for code, name, atype in ACCOUNT_SEED:
        db.add(LedgerAccount(ledger_id=ledger.id, code=code, name=name, account_type=atype, is_leaf=True))
    db.flush()
    return {"status": "ok", "account_id": account.id}

## 验证矩阵

| 场景 | 操作 | 结果 | 原因 |
|------|------|------|------|
| Agent 直写 ORM | `db.add(Product(...))` | ❌ 拦截 | `before_flush`，无令牌 |
| Agent 直写 SQL | `db.execute(text("DELETE..."))` | ❌ 拦截 | `before_execute`，无令牌 |
| Agent 合规 API | `POST /api/products` | ✅ 通过 | Gateway → 令牌 → 放行 |
| Agent 分析查询 | `db.execute("SELECT ... JOIN ...")` | ✅ 通过 | SELECT 自由 |
| Agent 删表 | `db.execute("DROP TABLE ...")` | ❌ 拦截 | DDL 红线，有令牌也不行 |
| User 前端操作 | `POST /api/sales` | ✅ 通过 | Gateway 对 User 放行 |
| 启动 init_db | `init_db()` | ✅ 通过 | `_maintenance_mode=True` |
| 只读改属性 | `GET` 中 `obj.name="x"` → flush | ❌ 拦截 | auto-flush 陷阱，提示修正 |
