## v1事件总线的不足

1. **无条件触发**：所有handler都会执行，无法按account_id/场景过滤
2. **无优先级**：handler执行顺序依赖注册顺序，不够明确
3. **无中间件**：无法在事件处理前后插入横切逻辑（如性能监控、不变量校验）
4. **无错误隔离**：一个handler异常中断所有后续handler

## v2事件总线

```python
# events.py — v2：支持优先级、条件过滤、中间件链
from collections import defaultdict
from dataclasses import dataclass, field
from typing import Callable, Any
import logging

logger = logging.getLogger("inventory")


@dataclass(order=True)
class HandlerEntry:
    """事件处理器条目 — 支持优先级排序"""
    sort_priority: int
    handler: Callable = field(compare=False)
    name: str = field(compare=False, default="")
    condition: Callable | None = field(compare=False, default=None)


class EventBus:
    """v2事件总线 — 支持中间件链、优先级、条件过滤"""

    def __init__(self):
        self._handlers: dict[str, list[HandlerEntry]] = defaultdict(list)
        self._middleware: list[Callable] = []

    def on(self, event_name: str, priority: int = 100, condition: Callable = None, name: str = ""):
        """注册事件处理器（装饰器）

        Args:
            event_name: 事件名
            priority: 优先级（数字越小越先执行）
            condition: 条件函数，接收 **kwargs，返回 bool
            name: 处理器名称（调试用）
        """
        def decorator(fn: Callable):
            entry = HandlerEntry(
                sort_priority=priority, handler=fn,
                name=name or fn.__name__, condition=condition
            )
            self._handlers[event_name].append(entry)
            self._handlers[event_name].sort()
            return fn
        return decorator

    def use(self, middleware: Callable):
        """注册中间件"""
        self._middleware.append(middleware)

    def emit(self, event_name: str, **kwargs):
        """触发事件"""
        entries = self._handlers.get(event_name, [])
        if not entries:
            return
        active_handlers = []
        for entry in entries:
            if entry.condition and not entry.condition(**kwargs):
                continue
            active_handlers.append(entry.handler)
        if not active_handlers:
            return
        logger.debug(f"事件触发: {event_name}，活跃处理器: {[h.__name__ for h in active_handlers]}")
        self._run_middleware(event_name, active_handlers, kwargs, 0)

    def _run_middleware(self, event_name, handlers, kwargs, mw_index):
        if mw_index < len(self._middleware):
            def next_mw():
                self._run_middleware(event_name, handlers, kwargs, mw_index + 1)
            self._middleware[mw_index](event_name, handlers, kwargs, next_mw)
        else:
            for handler in handlers:
                handler(**kwargs)

    def get_handlers(self, event_name: str) -> list[dict]:
        """调试用：查看某事件的所有处理器"""
        return [
            {"name": e.name, "priority": e.sort_priority,
             "handler": e.handler.__name__, "has_condition": e.condition is not None}
            for e in self._handlers.get(event_name, [])
        ]


# 全局单例 + 便捷函数
bus = EventBus()

def on(event_name, priority=100, condition=None, name=""):
    return bus.on(event_name, priority=priority, condition=condition, name=name)

def emit(event_name, **kwargs):
    return bus.emit(event_name, **kwargs)

def use(middleware):
    return bus.use(middleware)

def get_handlers(event_name):
    return bus.get_handlers(event_name)
```

## 事件中间件

```python
# middleware.py
import time
import logging

logger = logging.getLogger("inventory")


def logging_middleware(event_name, handlers, kwargs, next_mw):
    """日志中间件：记录事件触发和耗时"""
    start = time.time()
    logger.info(f"[事件] {event_name} 开始，{len(handlers)}个处理器")
    next_mw()
    elapsed = (time.time() - start) * 1000
    logger.info(f"[事件] {event_name} 完成，耗时{elapsed:.1f}ms")


def invariant_check_middleware(event_name, handlers, kwargs, next_mw):
    """不变量校验中间件：写操作事件后自动校验三大不变量"""
    next_mw()
    if event_name.endswith((".created", ".updated", ".deleted", ".cancelled")):
        db = kwargs.get("db")
        account_id = kwargs.get("account_id")
        if db and account_id:
            from utils import verify_invariants
            result = verify_invariants(db, account_id)
            if not result["ok"]:
                logger.warning(f"[不变量] {event_name} 后违规: {result['violations']}")


def register_middleware():
    """注册中间件（app启动时调用）"""
    from events import use
    use(logging_middleware)
    use(invariant_check_middleware)
```

## 副作用注册（利用优先级和条件）

```python
# handlers.py
from events import on
from crud.linkage import (
    cost_deduct_inventory, cost_restore_inventory, cost_update_inventory,
    sale_deduct_inventory, sale_restore_inventory,
    sale_create_income, sale_delete_income,
)
from crud.base import _log
from utils import update_project_summary


# ── 销售单事件 ──

@on("sale_order.created", priority=10, name="log")
def log_sale_create(db, account_id, order, operator):
    _log(db, account_id, "create", "sale_order", order.id,
         f"创建销售单 {order.order_no}", operator=operator)

@on("sale_order.created", priority=50, name="deduct_inventory",
    condition=lambda order, **kw: bool(order.deduct_inventory) and order.project_id is None)
def deduct_inventory_for_retail(db, account_id, order, operator):
    """零售扣库存 — 仅当 deduct_inventory=True 且无项目关联时"""
    sale_deduct_inventory(db, account_id, order, operator)

@on("sale_order.created", priority=50, name="create_income",
    condition=lambda order, **kw: order.project_id is not None)
def create_project_income_from_sale(db, account_id, order, operator):
    """自动生成项目收入 — 仅当关联项目时"""
    sale_create_income(db, account_id, order, operator)

@on("sale_order.created", priority=90, name="recalc_summary",
    condition=lambda order, **kw: order.project_id is not None)
def recalc_project_for_sale(db, account_id, order, operator):
    update_project_summary(db, order.project_id)


# ── 销售单取消/删除事件 ──

@on("sale_order.cancelled", priority=50, name="restore_inventory",
    condition=lambda order, **kw: bool(order.deduct_inventory) and order.project_id is None)
def restore_inventory_on_cancel(db, account_id, order, operator):
    sale_restore_inventory(db, account_id, order, operator)

@on("sale_order.cancelled", priority=50, name="delete_income",
    condition=lambda order, **kw: order.project_id is not None)
def delete_project_income_on_cancel(db, account_id, order, operator):
    sale_delete_income(db, account_id, order.id, operator)


# ── 项目成本事件 ──

@on("project_cost.created", priority=10, name="log")
def log_cost_create(db, account_id, cost, operator):
    _log(db, account_id, "create", "project_cost", cost.id,
         f"添加成本:{cost.cost_type} {cost.amount}", operator=operator)

@on("project_cost.created", priority=30, name="deduct_inventory",
    condition=lambda cost, **kw: cost.cost_type == "材料" and cost.product_id is not None)
def deduct_inventory_for_material(db, account_id, cost, operator):
    cost_deduct_inventory(db, account_id, cost, operator)

@on("project_cost.created", priority=90, name="recalc_summary")
def recalc_project_for_cost(db, account_id, cost, operator):
    update_project_summary(db, cost.project_id)
```
