## Pinia Store

```javascript
// stores/enums.js — Pinia枚举store
import { defineStore } from 'pinia'
import { api } from '../api/common'

export const useEnumsStore = defineStore('enums', {
  state: () => ({
    values: null,  // { expense_categories: [...], ... }
    labels: null,  // { order_status: { pending: '待处理', ... }, ... }
    loaded: false,
  }),
  getters: {
    getValues: (state) => (type) => state.values?.[type] || [],
    getLabel: (state) => (type, value) => state.labels?.[type]?.[value] || value,
    paymentStatusOptions(state) {
      return (state.values?.payment_status || []).map(v => ({
        value: v, label: state.labels?.payment_status?.[v] || v
      }))
    },
    orderStatusOptions(state) {
      return (state.values?.order_status || []).map(v => ({
        value: v, label: state.labels?.order_status?.[v] || v
      }))
    },
  },
  actions: {
    async loadEnums() {
      if (this.loaded) return
      const data = await api.getEnums()
      this.values = data.values
      this.labels = data.labels
      this.loaded = true
    },
  },
})
```

```javascript
// stores/projects.js — Pinia项目store
import { defineStore } from 'pinia'
import projectsApi from '../api/projects'

export const useProjectsStore = defineStore('projects', {
  state: () => ({
    projects: [],
    managedProjects: [],
    currentDetail: null,
    loading: false,
  }),
  actions: {
    async fetchProjects() {
      this.loading = true
      try {
        this.projects = await projectsApi.getProjects()
      } finally {
        this.loading = false
      }
    },
    async fetchProjectDetail(id) {
      this.currentDetail = await projectsApi.getProjectDetail(id)
      return this.currentDetail
    },
    clearDetail() {
      this.currentDetail = null
    },
  },
})
```

## API层拆分 + 响应校验

```javascript
// api/projects.js
import api from './index'

// 轻量级运行时校验
function validateResponse(data, requiredFields, apiName) {
  if (Array.isArray(data)) return data
  for (const field of requiredFields) {
    if (!(field in data)) {
      console.warn(`[API校验] ${apiName} 返回数据缺少字段: ${field}`)
    }
  }
  return data
}

export default {
  getProjects: () => api.get('/projects/'),
  getProjectList: () => api.get('/projects/list'),
  getProjectDetail: (id) => api.get(`/projects/${id}`).then(d =>
    validateResponse(d, ['id', 'name', 'total_income', 'total_cost', 'profit'], 'getProjectDetail')
  ),
  createProject: (data) => api.post('/projects/', data),
  updateProject: (id, data) => api.put(`/projects/${id}`, data),
  deleteProject: (id) => api.delete(`/projects/${id}`),
  getProjectCosts: (params) => api.get('/costs/', { params }),
  createProjectCost: (data) => api.post('/costs/', data),
  updateProjectCost: (id, data) => api.put(`/costs/${id}`, data),
  deleteProjectCost: (id) => api.delete(`/costs/${id}`),
  getProjectIncomes: (params) => api.get('/costs/incomes/', { params }),
  createProjectIncome: (data) => api.post('/costs/incomes/', data),
  updateProjectIncome: (id, data) => api.put(`/costs/incomes/${id}`, data),
  deleteProjectIncome: (id) => api.delete(`/costs/incomes/${id}`),
}
```
