## 为什么需要领域模型

当前业务逻辑直接操作ORM模型（`models.SaleOrder`），导致：
- 业务规则散落在crud/routers/linkage各处
- 不变量校验只在`verify_invariants`中，不是内建的
- 新增业务规则时不知道该改哪里

**领域模型**将业务规则和不变量封装在Domain类中，ORM模型只做持久化。

## Money值对象

```python
# domain/money.py
from decimal import Decimal, ROUND_HALF_UP


class Money:
    """金额值对象 — 封装Decimal运算，杜绝float泄漏

    规则：
    1. 所有金额运算必须通过Money类，禁止直接使用float
    2. 运算结果自动四舍五入到2位小数
    3. 比较使用精确比较，不容差
    """

    def __init__(self, amount: Decimal | str | int | float = Decimal('0')):
        if isinstance(amount, float):
            amount = Decimal(str(round(amount, 10)))  # float必须经字符串转换
        elif not isinstance(amount, Decimal):
            amount = Decimal(str(amount))
        self._amount = amount.quantize(Decimal('0.01'), rounding=ROUND_HALF_UP)

    @property
    def amount(self) -> Decimal:
        return self._amount

    def to_float(self) -> float:
        """仅在ORM写入时使用"""
        return float(self._amount)

    def __add__(self, other: 'Money') -> 'Money':
        return Money(self._amount + other._amount)

    def __sub__(self, other: 'Money') -> 'Money':
        return Money(self._amount - other._amount)

    def __mul__(self, factor: Decimal | int) -> 'Money':
        return Money(self._amount * Decimal(str(factor)))

    def __eq__(self, other) -> bool:
        if isinstance(other, Money):
            return self._amount == other._amount
        return NotImplemented

    def __lt__(self, other: 'Money') -> bool:
        return self._amount < other._amount

    def __le__(self, other: 'Money') -> bool:
        return self._amount <= other._amount

    def __repr__(self) -> str:
        return f"Money({self._amount})"

    def __str__(self) -> str:
        return f"¥{self._amount}"

    @classmethod
    def zero(cls) -> 'Money':
        return cls('0')

    @classmethod
    def sum(cls, amounts: list['Money']) -> 'Money':
        total = Decimal('0')
        for m in amounts:
            total += m._amount
        return cls(total)
```

## 领域模型基类

```python
# domain/base.py
from abc import ABC, abstractmethod
from typing import TypeVar, Generic

TModel = TypeVar('TModel')


class DomainModel(ABC, Generic[TModel]):
    """领域模型基类 — 封装业务规则和不变量，与ORM模型解耦"""

    @classmethod
    @abstractmethod
    def from_orm(cls, orm_obj: TModel) -> 'DomainModel':
        """ORM对象 → Domain对象"""
        ...

    @abstractmethod
    def validate(self) -> list[str]:
        """不变量校验，返回违规列表（空列表=通过）"""
        ...
```

## 销售单领域模型（含状态机）

```python
# domain/sale_order.py
from __future__ import annotations
from dataclasses import dataclass, field
from decimal import Decimal
from typing import Optional
from domain.base import DomainModel
from domain.money import Money


VALID_TRANSITIONS = {
    "pending": {"completed", "cancelled"},
    "completed": {"cancelled"},
    "cancelled": {"completed"},  # 恢复
}


@dataclass
class SaleOrderLine:
    product_id: int
    quantity: int
    unit_price: Money
    tax_rate: Decimal
    total_price: Money


@dataclass
class SaleOrderDomain(DomainModel):
    """销售单领域模型 — 封装业务规则

    核心规则：
    1. 状态机：只允许合法的状态转换
    2. 金额一致：total_price = sum(items.total_price)
    3. 零售规则：deduct_inventory=True时必须有items且无project_id
    4. 项目规则：project_id存在时禁止deduct_inventory
    """

    id: int = 0
    account_id: int = 0
    order_no: str = ""
    customer_id: Optional[int] = None
    project_id: Optional[int] = None
    project_name: str = ""
    status: str = "pending"
    payment_status: str = "pending"
    payment_method: str = "company"
    has_invoice: bool = False
    deduct_inventory: bool = False
    sale_date: str = ""
    notes: str = ""
    total_price: Money = Money.zero()
    items: list[SaleOrderLine] = field(default_factory=list)

    # ── 状态机 ──

    def can_transition_to(self, new_status: str) -> bool:
        return new_status in VALID_TRANSITIONS.get(self.status, set())

    def transition_to(self, new_status: str):
        if not self.can_transition_to(new_status):
            raise ValueError(
                f"非法状态转换: {self.status} → {new_status}，"
                f"合法目标: {VALID_TRANSITIONS.get(self.status, set())}"
            )
        self.status = new_status

    # ── 业务规则 ──

    def is_retail(self) -> bool:
        return self.deduct_inventory and self.project_id is None

    def is_project_sale(self) -> bool:
        return self.project_id is not None

    def should_deduct_inventory(self) -> bool:
        return self.is_retail() and self.status == "completed"

    # ── 不变量校验 ──

    def validate(self) -> list[str]:
        violations = []
        if self.project_id and self.deduct_inventory:
            violations.append("项目销售单不支持直接扣库存")
        if self.deduct_inventory and not self.items:
            violations.append("扣库存销售单必须有商品明细")
        if self.items:
            calc_total = Money.sum([item.total_price for item in self.items])
            if calc_total != self.total_price:
                violations.append(f"金额不一致: total={self.total_price}, 明细={calc_total}")
        return violations

    # ── ORM转换 ──

    @classmethod
    def from_orm(cls, orm_obj) -> 'SaleOrderDomain':
        items = [
            SaleOrderLine(
                product_id=item.product_id, quantity=item.quantity,
                unit_price=Money(item.unit_price),
                tax_rate=Decimal(str(item.tax_rate)) if item.tax_rate else Decimal('0'),
                total_price=Money(item.total_price),
            )
            for item in (orm_obj.items or [])
        ]
        return cls(
            id=orm_obj.id, account_id=orm_obj.account_id,
            order_no=orm_obj.order_no, customer_id=orm_obj.customer_id,
            project_id=orm_obj.project_id, project_name=orm_obj.project_name or "",
            status=orm_obj.status, payment_status=orm_obj.payment_status,
            payment_method=orm_obj.payment_method, has_invoice=orm_obj.has_invoice,
            deduct_inventory=bool(orm_obj.deduct_inventory) if orm_obj.deduct_inventory else False,
            sale_date=str(orm_obj.sale_date) if orm_obj.sale_date else "",
            notes=orm_obj.notes or "", total_price=Money(orm_obj.total_price),
            items=items,
        )
```

## Repository升级（配合Domain模型）

```python
# repositories/order_repo.py — 配合领域模型的Repository
from sqlalchemy.orm import Session
import models
from domain.sale_order import SaleOrderDomain


class SaleOrderRepo:
    """销售单数据访问 — ORM↔Domain双向转换的唯一入口"""

    @staticmethod
    def get_domain(db: Session, account_id: int, order_id: int) -> SaleOrderDomain | None:
        orm = db.query(models.SaleOrder).filter(
            models.SaleOrder.account_id == account_id,
            models.SaleOrder.id == order_id
        ).first()
        if not orm:
            return None
        return SaleOrderDomain.from_orm(orm)

    @staticmethod
    def update_from_domain(db: Session, domain: SaleOrderDomain):
        orm = db.query(models.SaleOrder).filter(models.SaleOrder.id == domain.id).first()
        if not orm:
            return
        orm.status = domain.status
        orm.payment_status = domain.payment_status
        orm.total_price = domain.total_price.to_float()
        orm.project_id = domain.project_id
        orm.deduct_inventory = domain.deduct_inventory
        orm.notes = domain.notes
        db.flush()

    @staticmethod
    def save_new(db: Session, domain: SaleOrderDomain) -> int:
        orm = models.SaleOrder(
            account_id=domain.account_id, order_no=domain.order_no,
            customer_id=domain.customer_id, project_id=domain.project_id,
            total_price=domain.total_price.to_float(),
        )
        db.add(orm)
        db.flush()
        return orm.id
```

## Domain模型在Command中的使用

```python
# commands/sale_commands.py — 使用Domain模型后的Handler

@register(CancelSaleOrderCommand)
class CancelSaleOrderHandler(CommandHandler):
    def handle(self, cmd: CancelSaleOrderCommand, db: Session):
        # 1. 加载Domain模型
        domain = SaleOrderRepo.get_domain(db, cmd.account_id, cmd.order_id)
        if not domain:
            raise ValueError("销售单不存在")

        # 2. 业务规则校验（状态机）
        domain.transition_to("cancelled")  # 非法转换自动抛异常

        # 3. 不变量校验
        violations = domain.validate()
        if violations:
            raise ValueError(f"不变量违反: {violations}")

        # 4. 持久化
        SaleOrderRepo.update_from_domain(db, domain)

        # 5. 事件触发
        emit("sale_order.cancelled", db=db, account_id=cmd.account_id,
             order_id=cmd.order_id, operator=cmd.operator)

        return domain
```
