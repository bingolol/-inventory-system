## 为什么需要Command模式

当前 `update_sale_order()` 有近200行代码，混合了6种不同的业务场景：
- items全量替换（行415-488）
- 状态切换 completed↔cancelled（行500-519）
- project_id变更（行522-531）
- total_price变更+差额分配（行534-547）
- 普通字段更新（行492-494）
- items归零自动删除（行431-438）

每次新增场景都要修改这个巨型函数，任何修改都可能破坏其他场景。

**Command模式**将每种操作封装为独立的类，路由层只需 `dispatch(command)`。

## Command基类 + Dispatcher

```python
# commands/base.py
from __future__ import annotations
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any, Type, TypeVar, Callable
from sqlalchemy.orm import Session
import logging

logger = logging.getLogger("inventory")

TResult = TypeVar("TResult")


@dataclass
class Command(ABC):
    """命令基类 — 每个业务操作对应一个Command子类"""
    account_id: int
    operator: str = "user"


class CommandHandler(ABC):
    """处理器基类"""
    @abstractmethod
    def handle(self, cmd: Command, db: Session) -> Any:
        raise NotImplementedError


# ── 命令调度器 ──
_registry: dict[Type[Command], CommandHandler] = {}


def register(command_type: Type[Command]):
    """装饰器：注册命令处理器"""
    def decorator(handler_cls: Type[CommandHandler]):
        _registry[command_type] = handler_cls()
        return handler_cls
    return decorator


def dispatch(cmd: Command, db: Session) -> Any:
    """调度命令到对应处理器（在 unit_of_work 内调用）"""
    handler = _registry.get(type(cmd))
    if not handler:
        raise ValueError(f"未注册的命令类型: {type(cmd).__name__}")
    logger.debug(f"调度命令: {type(cmd).__name__}")
    return handler.handle(cmd, db)


def get_registered_commands() -> dict[str, str]:
    """调试用：查看所有已注册的命令"""
    return {cmd.__name__: handler.__class__.__name__ for cmd, handler in _registry.items()}
```

## 销售单命令示例

```python
# commands/sale_commands.py
from dataclasses import dataclass
from decimal import Decimal
from sqlalchemy.orm import Session
import models, schemas
from commands.base import Command, CommandHandler, register
from crud.base import _generate_order_no, _log
from crud.products import get_product
from events import emit


@dataclass
class CreateSaleOrderCommand(Command):
    data: schemas.SaleOrderCreate


@dataclass
class CancelSaleOrderCommand(Command):
    order_id: int


@dataclass
class RestoreSaleOrderCommand(Command):
    order_id: int


@dataclass
class DeleteSaleOrderCommand(Command):
    order_id: int


@dataclass
class UpdateSaleOrderItemsCommand(Command):
    """items全量替换"""
    order_id: int
    items: list
    total_price: Decimal | None = None


@dataclass
class UpdateSaleOrderFieldsCommand(Command):
    """普通字段更新（不含items）"""
    order_id: int
    updates: dict


@dataclass
class UpdateSaleOrderProjectCommand(Command):
    """project_id变更"""
    order_id: int
    new_project_id: int | None


@dataclass
class UpdateSaleOrderTotalPriceCommand(Command):
    """只改总额不改明细"""
    order_id: int
    total_price: Decimal


@register(CreateSaleOrderCommand)
class CreateSaleOrderHandler(CommandHandler):
    def handle(self, cmd: CreateSaleOrderCommand, db: Session):
        data = cmd.data
        if not data.items:
            raise ValueError("销售单至少包含1个商品")
        order_no = _generate_order_no(db, "SO")
        order = models.SaleOrder(
            account_id=cmd.account_id, order_no=order_no,
            customer_id=data.customer_id, project_id=data.project_id,
            project_name=data.project_name, has_invoice=data.has_invoice,
            payment_method=data.payment_method, payment_status="pending",
            deduct_inventory=data.deduct_inventory,
            sale_date=data.sale_date, notes=data.notes, total_price=0
        )
        db.add(order)
        db.flush()
        total = Decimal('0')
        for item_data in data.items:
            product = get_product(db, cmd.account_id, item_data.product_id)
            if not product:
                raise ValueError(f"商品不存在: ID={item_data.product_id}")
            line_total = Decimal(str(item_data.quantity)) * Decimal(str(item_data.unit_price))
            item = models.SaleItem(
                order_id=order.id, product_id=item_data.product_id,
                quantity=item_data.quantity, unit_price=item_data.unit_price,
                tax_rate=item_data.tax_rate, total_price=float(line_total)
            )
            db.add(item)
            total += line_total
        order.total_price = float(round(total, 2))
        emit("sale_order.created", db=db, account_id=cmd.account_id,
             order=order, operator=cmd.operator)
        return order


@register(CancelSaleOrderCommand)
class CancelSaleOrderHandler(CommandHandler):
    def handle(self, cmd: CancelSaleOrderCommand, db: Session):
        order = db.query(models.SaleOrder).filter(
            models.SaleOrder.id == cmd.order_id,
            models.SaleOrder.account_id == cmd.account_id
        ).first()
        if not order:
            raise ValueError("销售单不存在")
        if order.status == "cancelled":
            return order  # 幂等
        order.status = "cancelled"
        emit("sale_order.cancelled", db=db, account_id=cmd.account_id,
             order=order, operator=cmd.operator)
        return order


@register(DeleteSaleOrderCommand)
class DeleteSaleOrderHandler(CommandHandler):
    def handle(self, cmd: DeleteSaleOrderCommand, db: Session):
        order = db.query(models.SaleOrder).filter(
            models.SaleOrder.id == cmd.order_id,
            models.SaleOrder.account_id == cmd.account_id
        ).first()
        if not order:
            raise ValueError("销售单不存在")
        emit("sale_order.deleted", db=db, account_id=cmd.account_id,
             order=order, operator=cmd.operator)
        db.delete(order)
        return None
```

## Router层（极简化）

```python
# routers/sales.py — 改造后
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.orm import Session
from database import get_db
from account_dep import get_account_id, get_operator
from uow import unit_of_work
import schemas
from commands.base import dispatch
from commands.sale_commands import (
    CreateSaleOrderCommand, CancelSaleOrderCommand,
    DeleteSaleOrderCommand, UpdateSaleOrderItemsCommand,
    UpdateSaleOrderFieldsCommand, UpdateSaleOrderProjectCommand,
)

router = APIRouter()


@router.post("/", response_model=schemas.SaleOrderOut)
def create_sale(data: schemas.SaleOrderCreate,
                account_id=Depends(get_account_id), operator=Depends(get_operator),
                db=Depends(get_db)):
    cmd = CreateSaleOrderCommand(account_id=account_id, operator=operator, data=data)
    try:
        with unit_of_work(db):
            order = dispatch(cmd, db)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    db.refresh(order)
    return _build_sale_out(order)


@router.put("/{order_id}/cancel")
def cancel_sale(order_id, account_id=Depends(get_account_id),
                operator=Depends(get_operator), db=Depends(get_db)):
    cmd = CancelSaleOrderCommand(account_id=account_id, operator=operator, order_id=order_id)
    try:
        with unit_of_work(db):
            dispatch(cmd, db)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    return {"message": "销售单已取消"}


@router.put("/{order_id}", response_model=schemas.SaleOrderOut)
def update_sale(order_id, data: schemas.SaleOrderUpdate,
                account_id=Depends(get_account_id), operator=Depends(get_operator),
                db=Depends(get_db)):
    try:
        with unit_of_work(db):
            if data.items is not None:
                cmd = UpdateSaleOrderItemsCommand(
                    account_id=account_id, operator=operator,
                    order_id=order_id, items=data.items, total_price=data.total_price)
            elif data.project_id is not None:
                cmd = UpdateSaleOrderProjectCommand(
                    account_id=account_id, operator=operator,
                    order_id=order_id, new_project_id=data.project_id)
            else:
                cmd = UpdateSaleOrderFieldsCommand(
                    account_id=account_id, operator=operator,
                    order_id=order_id, updates=data.model_dump(exclude_unset=True))
            dispatch(cmd, db)
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    order = db.query(models.SaleOrder).filter(models.SaleOrder.id == order_id).first()
    return _build_sale_out(order)
```
